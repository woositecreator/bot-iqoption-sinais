Main : justify-content {
    flex-start: padrao inicia os itens,
    no inicio do container


    flex-end: alinha os item no final do container


    center: alinha os item no centro do container


    space-betwen: distribui os items no container

    space-around: igual o space-betwen mais cria um padding
}

Cross : align-items {
    stretch: padrao faz com que os items crescam igualmente

    flex-start: alinha os items para o comeco do container

    flex-send: alinha os items para final do container
    
    center: alinha os items no centro do container

    space-betwen: distribui os items no container

    space-around: igual o space-betwen mais cria um padding
}

flex-flow {
    nowrap: impede que o box se quebrado de linha
    wrap: quebra a linha assim que um dois dois nao couber nela
    wrap-resver: igual ao wrap mais quebra em cima
}

Items : order: muda a ordem
        flex: aumenta o tamnho baseado no Main
        flex-basis: aumenta a altura baseada no Main
